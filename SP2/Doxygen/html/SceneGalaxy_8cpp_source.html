503 bool SceneGalaxy::CheckCollision(Vector3 A, Vector3 B)
504 {
505  float x = 15.f, y = 15.f, z = 500.f;
506  if (A.x > B.x - x && A.x < B.x + x
507  && A.y > B.y && A.y < B.y + y
508  && A.z> B.z - z && A.z < B.z + z)
509  return true;
510  return false;
511 }

513 /****************************************************************************/
522 /****************************************************************************/
523 int SceneGalaxy::getMagnitude(Vector3 A, Vector3 B)
524 {
525  // A camera, B target
526  int totalX = (A.x - B.x) * (A.x - B.x);
527  int totalY = (A.y - B.y) * (A.y - B.y);
528  int totalZ = (A.z - B.z) * (A.z - B.z);
529  int magnitude = totalX + totalY + totalZ;
530  return sqrt(magnitude);
531 }
